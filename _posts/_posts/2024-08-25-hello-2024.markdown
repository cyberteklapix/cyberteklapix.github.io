``
#include <iostream>
#include <math.h>
using namespace std;
bool isPrime(int n){
    if(n<=1) return 0;
    int sqr = (int)sqrt(1.0*n);
    for(int i=2;i<=sqr;++i)
        if(n%i==0) return 0;
    return 1;
}
int main()
{
    int n, Count = 0;
    cin >> n;
    for(int i=3;i+2<=n;i+=2)
        if(isPrime(i)==1 && isPrime(i+2)==1)
            Count++;
    cout << Count << endl;
}
